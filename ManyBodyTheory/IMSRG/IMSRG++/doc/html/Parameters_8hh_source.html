119 void Parameters::ParseCommandLineArgs(int argc, char** argv)
120 {
121  cout << "==================== Parameters (defaults set in Parameters.hh) ===================" << endl;
122  for (int iarg=1; iarg<argc; ++iarg)
123  {
124  string arg = argv[iarg];
125  if (arg=="help" or arg=="-help" or arg=="--help")
126  {
127  cout << "\nUsage:\n\timsrg++ option1=variable1 option2=variable2...\n" << endl;
128  cout << "At minimum, the 2bme file is required.\n" << endl;
129  PrintOptions();
130  help_mode = true;
131  return;
132  }
133  size_t pos = arg.find("=");
134  string var = arg.substr(0,pos);
135  string val = arg.substr(pos+1);
136  if (string_par.find(var) != string_par.end() )
137  {
138  if (val.size() > 0)
139  istringstream(val) >> string_par[var];
140  cout << var << " => " << string_par[var] << endl;
141  }
142  else if (double_par.find(var) != double_par.end() )
143  {
144  if (val.size() > 0)
145  istringstream(val) >> double_par[var];
146  cout << var << " => " << double_par[var] << endl;
147  }
148  else if (int_par.find(var) != int_par.end() )
149  {
150  if (val.size() > 0)
151  istringstream(val) >> int_par[var];
152  cout << var << " => " << int_par[var] << endl;
153  }
154  else if (vec_par.find(var) != vec_par.end() )
155  {
156  if (val.size() > 0)
157  {
158  istringstream ss(val);
159  string tmp;
160  while( getline(ss,tmp,',')) vec_par[var].push_back(tmp);
161  }
162  cout << var << " => ";
163  for (auto x : vec_par[var]) cout << x << ",";
164  cout << endl;
165  }
166  else
167  {
168  cout << "Unkown parameter: " << var << " => " << val << endl;
169  }
170 
171  }
172  if (string_par["flowfile"]=="default") string_par["flowfile"] = DefaultFlowFile();
173  if (string_par["intfile"]=="default") string_par["intfile"] = DefaultIntFile();
174  cout << "====================================================================================" << endl;
175 }
