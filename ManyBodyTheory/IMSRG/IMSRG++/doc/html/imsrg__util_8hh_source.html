139  template <typename T, typename... Args>
140  T VectorUnion(const T& v1, const T& v2, Args... args)
141  {
142  T vec(v1.size()+v2.size());
143  copy(v1.begin(),v1.end(),vec.begin());
144  copy(v2.begin(),v2.end(),vec.begin()+v1.size());
145  return VectorUnion(vec, args...);
